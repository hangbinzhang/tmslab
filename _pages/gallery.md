<!--
  Auto-discovery: every .jpg/.jpeg/.png/.webp file in
  /assets/img/gallery/ is picked up automatically at build time.
  Drop a file in → it appears on next rebuild. Remove a file →
  it disappears. No YAML to maintain.

  Filename convention (see assets/img/gallery/README.md):
    YYYY-MM-DD_short_description.jpg

  Files MATCHING the convention:
    - date prefix parsed out, remainder becomes the hover caption
      (underscores/hyphens → spaces, sentence-cased)
    - date rendered as a "Mon YYYY" label below the caption
    - sorted newest-first by date

  Files NOT matching the convention:
    - no caption overlay at all (photo only)
    - sorted after all dated photos, in reverse-alphabetical order

  Filtered out: responsive variants generated by jekyll-imagemagick
  (basenames ending in -480 / -800 / -1400). Only the originals show.
-->

<div class="tmslab-gallery-header">
  <p class="tmslab-gallery-subtitle">A collection of moments from the lab — retreats, events, conference travel, and the quiet work between experiments.</p>
</div>

{%- assign gallery_images = site.static_files | where_exp: "f", "f.path contains '/assets/img/gallery/'" -%}
{%- assign gallery_images = gallery_images | where_exp: "f", "f.extname == '.jpg' or f.extname == '.jpeg' or f.extname == '.png' or f.extname == '.webp'" -%}

{%- comment -%}
  Build a list of `sort_date|path|caption|date_label` strings so we
  can sort the whole set by date descending in one pass. Undated
  files get sort_date "0000-00-00" so they land at the end after
  the reverse. Imagemagick variants (-480 / -800 / -1400) are
  skipped at this stage, not shown at all.
{%- endcomment -%}

{%- assign valid_items = "" | split: "" -%}

{%- for image in gallery_images -%}
  {%- assign segs = image.basename | split: "-" -%}
  {%- assign last_seg = segs | last -%}
  {%- unless last_seg == "480" or last_seg == "800" or last_seg == "1400" -%}

    {%- assign basename = image.basename -%}
    {%- assign has_date = false -%}
    {%- assign date_label = "" -%}
    {%- assign sort_date = "0000-00-00" -%}
    {%- assign raw_caption = basename -%}

    {%- comment -%} Probe for a YYYY-MM-DD_ prefix. {%- endcomment -%}
    {%- assign first_ten = basename | slice: 0, 10 -%}
    {%- assign rest = basename | slice: 10, 200 -%}
    {%- if first_ten.size == 10 and rest.size > 0 -%}
      {%- assign d1 = first_ten | slice: 4, 1 -%}
      {%- assign d2 = first_ten | slice: 7, 1 -%}
      {%- assign sep = rest | slice: 0, 1 -%}
      {%- if d1 == "-" and d2 == "-" and sep == "_" -%}
        {%- assign has_date = true -%}
        {%- assign sort_date = first_ten -%}
        {%- assign raw_caption = rest | slice: 1, 200 -%}
        {%- assign year = first_ten | slice: 0, 4 -%}
        {%- assign month = first_ten | slice: 5, 2 -%}
        {%- case month -%}
          {%- when "01" -%}{%- assign month_name = "Jan" -%}
          {%- when "02" -%}{%- assign month_name = "Feb" -%}
          {%- when "03" -%}{%- assign month_name = "Mar" -%}
          {%- when "04" -%}{%- assign month_name = "Apr" -%}
          {%- when "05" -%}{%- assign month_name = "May" -%}
          {%- when "06" -%}{%- assign month_name = "Jun" -%}
          {%- when "07" -%}{%- assign month_name = "Jul" -%}
          {%- when "08" -%}{%- assign month_name = "Aug" -%}
          {%- when "09" -%}{%- assign month_name = "Sep" -%}
          {%- when "10" -%}{%- assign month_name = "Oct" -%}
          {%- when "11" -%}{%- assign month_name = "Nov" -%}
          {%- when "12" -%}{%- assign month_name = "Dec" -%}
          {%- else -%}{%- assign month_name = "" -%}
        {%- endcase -%}
        {%- assign date_label = month_name | append: " " | append: year -%}
      {%- endif -%}
    {%- endif -%}

    {%- comment -%}
      Sentence-case caption built with an accumulator — inline Liquid
      whitespace stripping collapses spaces inside a filter chain, so
      we do `capitalize` once on the joined, space-normalized string
      rather than on each word. Matches the "Lab retreat" style.
    {%- endcomment -%}
    {%- assign with_spaces = raw_caption | replace: "_", " " | replace: "-", " " -%}
    {%- assign words = with_spaces | split: " " -%}
    {%- assign caption = "" -%}
    {%- for w in words -%}
      {%- if w != "" -%}
        {%- if caption == "" -%}
          {%- assign caption = w -%}
        {%- else -%}
          {%- assign caption = caption | append: " " | append: w -%}
        {%- endif -%}
      {%- endif -%}
    {%- endfor -%}
    {%- assign caption = caption | capitalize -%}

    {%- capture meta -%}{{ sort_date }}|{{ image.path }}|{{ caption }}|{{ date_label }}{%- endcapture -%}
    {%- assign valid_items = valid_items | push: meta -%}

  {%- endunless -%}
{%- endfor -%}

{%- assign sorted_items = valid_items | sort | reverse -%}
{%- assign visible_count = sorted_items | size -%}

{% if visible_count == 0 %}
  <p class="tmslab-gallery-empty">Photos coming soon.</p>
{% else %}
  <div class="tmslab-gallery-grid" id="tmslab-gallery">
    {%- for item in sorted_items -%}
      {%- assign parts = item | split: "|" -%}
      {%- assign photo_path = parts[1] -%}
      {%- assign photo_caption = parts[2] -%}
      {%- assign photo_date = parts[3] -%}

      {%- if photo_date == blank -%}
        {%- comment -%}
          Filename didn't follow the YYYY-MM-DD_ convention —
          render the photo only, with no caption overlay. The
          lightbox still opens but shows no caption text either.
        {%- endcomment -%}
        <figure class="tmslab-gallery-item tmslab-gallery-item--no-caption" data-caption="" data-full="{{ photo_path | relative_url }}">
          <img src="{{ photo_path | relative_url }}" alt="" loading="lazy">
        </figure>
      {%- else -%}
        <figure class="tmslab-gallery-item" data-caption="{{ photo_caption | escape }}" data-full="{{ photo_path | relative_url }}">
          <img src="{{ photo_path | relative_url }}" alt="{{ photo_caption | escape }}" loading="lazy">
          <figcaption class="tmslab-gallery-caption">
            <span class="tmslab-gallery-caption-title">{{ photo_caption }}</span>
            <span class="tmslab-gallery-caption-date">{{ photo_date }}</span>
          </figcaption>
        </figure>
      {%- endif -%}
    {%- endfor -%}
  </div>

  <script src="{{ '/assets/js/gallery-lightbox.js' | relative_url }}" defer></script>
{% endif %}
